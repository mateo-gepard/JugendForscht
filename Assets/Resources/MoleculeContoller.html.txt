    <script>
        // === STATE ===
        let ws = null;
        let tutorialActive = false;
        let currentStep = 0;
        const totalSteps = 3;
        
        // === DOM ELEMENTS ===
        const statusEl = document.getElementById('status');
        const tutorialStatusEl = document.getElementById('tutorialStatus');
        const btnStart = document.getElementById('btnStart');
        const btnClose = document.getElementById('btnClose');
        const btnPrev = document.getElementById('btnPrev');
        const btnNext = document.getElementById('btnNext');
        const debugEl = document.getElementById('debug');
        
        // === DEBUG LOG ===
        function log(msg) {
            console.log(msg);
            const time = new Date().toLocaleTimeString();
            debugEl.innerHTML = `[${time}] ${msg}<br>` + debugEl.innerHTML;
            if (debugEl.children.length > 20) {
                debugEl.innerHTML = debugEl.innerHTML.split('<br>').slice(0, 20).join('<br>');
            }
        }
        
        // === WEBSOCKET ===
        function connect() {
            const wsUrl = 'ws://' + window.location.host + '/ws';
            log('Connecting to: ' + wsUrl);
            
            ws = new WebSocket(wsUrl);
            
            ws.onopen = () => {
                log('WebSocket OPEN');
                statusEl.textContent = '‚óè Verbunden';
                statusEl.className = 'connected';
            };
            
            ws.onclose = () => {
                log('WebSocket CLOSED - reconnecting in 2s');
                statusEl.textContent = '‚óè Getrennt';
                statusEl.className = 'disconnected';
                ws = null;
                setTimeout(connect, 2000);
            };
            
            ws.onerror = (e) => {
                log('WebSocket ERROR: ' + e.type);
            };
            
            ws.onmessage = (e) => {
                log('RECV: ' + e.data);
                try {
                    const data = JSON.parse(e.data);
                    handleMessage(data);
                } catch (err) {
                    log('Parse error: ' + err);
                }
            };
        }
        
        // === MESSAGE HANDLING ===
        function handleMessage(data) {
            if (data.type === 'tutorial') {
                // Update state from server
                if (data.status === 'started') {
                    tutorialActive = true;
                    currentStep = data.step || 1;
                } else if (data.status === 'closed') {
                    tutorialActive = false;
                    currentStep = 0;
                } else if (data.status === 'continued' || data.status === 'next') {
                    currentStep = data.step || (currentStep + 1);
                    if (currentStep > totalSteps) {
                        tutorialActive = false;
                        currentStep = 0;
                    }
                } else if (data.status === 'previous') {
                    currentStep = data.step || (currentStep - 1);
                    if (currentStep < 1) currentStep = 1;
                }
                updateUI();
            } else if (data.type === 'loaded') {
                statusEl.textContent = `‚óè Geladen: ${data.molecule} (${data.atoms} Atome)`;
            } else if (data.type === 'error') {
                statusEl.textContent = `‚óè Fehler: ${data.message}`;
            } else if (data.type === 'status') {
                statusEl.textContent = `‚óè ${data.message}`;
            }
        }
        
        // === SEND COMMANDS ===
        function send(obj) {
            if (ws && ws.readyState === WebSocket.OPEN) {
                const msg = JSON.stringify(obj);
                log('SEND: ' + msg);
                ws.send(msg);
                return true;
            } else {
                log('ERROR: WebSocket not connected');
                return false;
            }
        }
        
        function sendTutorial(action) {
            log('Button clicked: ' + action);
            send({ type: 'tutorial', action: action });
        }
        
        function loadMolecule(name) {
            if (send({ type: 'load', molecule: name })) {
                statusEl.textContent = `‚óè Lade ${name}...`;
            }
        }
        
        function searchMolecule() {
            const input = document.getElementById('searchInput');
            const name = input.value.trim();
            if (name) {
                loadMolecule(name);
                input.value = '';
            }
        }
        
        // === UI UPDATE ===
        function updateUI() {
            log('UI update: active=' + tutorialActive + ' step=' + currentStep);
            
            if (tutorialActive) {
                tutorialStatusEl.className = 'tutorial-status active';
                tutorialStatusEl.textContent = `Schritt ${currentStep} von ${totalSteps}`;
                btnStart.style.display = 'none';
                btnClose.style.display = 'block';
                btnPrev.style.display = 'block';
                btnNext.style.display = 'block';
                btnPrev.disabled = (currentStep <= 1);
                btnNext.textContent = (currentStep >= totalSteps) ? '‚úÖ Fertig' : 'Weiter ‚û°Ô∏è';
            } else {
                tutorialStatusEl.className = 'tutorial-status';
                tutorialStatusEl.textContent = 'Tutorial nicht gestartet';
                btnStart.style.display = 'block';
                btnClose.style.display = 'none';
                btnPrev.style.display = 'none';
                btnNext.style.display = 'none';
            }
        }
        
        // === INIT ===
        document.getElementById('searchInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') searchMolecule();
        });
        
        updateUI();
        connect();
    </script>
